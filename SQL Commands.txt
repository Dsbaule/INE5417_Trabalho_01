-- TABLE FORMATACAO

DROP TABLE FORMATACAO;

CREATE TABLE FORMATACAO
(
	NOME 		VARCHAR(20),
	TAMANHO		INT NOT NULL,
	FONTE		INT NOT NULL,
	CORFONTE	INT NOT NULL,
	CORFUNDO	INT NOT NULL,
	PRIMARY KEY (NOME)
);

-- TABLE DOCUMENTO

DROP TABLE DOCUMENTO;

CREATE TABLE DOCUMENTO
(
	NOME		VARCHAR(20),
	FILEPATH	VARCHAR(100) NOT NULL,
	USUARIO		VARCHAR(20) NOT NULL,
	FORMATACAO	VARCHAR(20) NOT NULL,
	PRIMARY KEY (NOME),
	FOREIGN KEY (USUARIO) REFERENCES USUARIO,
	FOREIGN KEY (FORMATACAO) REFERENCES FORMATACAO
);
